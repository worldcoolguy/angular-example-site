nuget.getstats
A Node.js package that gets a nuget package statistics.

Usage 


npm install nuget-getstats -S
                
var nugetStats = require('./index');
nugetStats
    .GetNugetPackageStats('printclassinstance')
    .then(data => console.log(data));

Output 

{{ '{' }}
    Title: 'Object and Type Extensions',
    ProjectUrl: 'https://harip.github.io/site/printclassinstance',
    Version: '3.2.0',
    TotalDownloads: 2306 
{{ '}' }}
